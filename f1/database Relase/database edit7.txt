create  PROCEDURE [dbo].[statment_item] @datee date, @datee2 date,@c nvarchar(50),@wareID nvarchar(50),@unite nvarchar(10)
as

select  '' as time, h.name_book as nam_doc,[no],d.purchase_hd_id as no_inv,d.qty*d.f_unite/@unite as qty,(select cost from wares where code_items=@c and id_ware=@wareID) as item_price,code_items,name_items,h.date_P as [date],h.vcs_name,h.vcs_code,name_unite from purchase_dt d left join purchase_hd h on d.purchase_hd_id=h.purchase_hd_id  where d.code_items=@c  and  h.date_P between @datee and @datee2 and h.id_ware=@wareID  
union all
